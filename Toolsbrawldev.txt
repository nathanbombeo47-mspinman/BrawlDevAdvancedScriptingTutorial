local tool = script.Parent
local handle = tool.Handle
local sound = tool.SoundBata2x

local holding = false
local WAIT_TIME = 0.2

tool.Equipped:Connect(function()
	print("this tool has been equipped")
	sound:Play()
end)

tool.Unequipped:Connect(function()
	print("this tool has been unequipped")
end)

tool.Activated:Connect(function()
	local part = Instance.new("Part", workspace)
	part.Position = Vector3.new(0, 50, 0)
	part.Size = Vector3.new(10, 10, 10)
	print("a new part has respawn")
	print("Sound Played")
	holding = true
	tool:ScaleTo(2)
	
	while holding do
		handle.Color =  Color3.fromRGB(math.random(0, 255), math.random(0, 255), math.random(0, 255))
		task.wait(WAIT_TIME)
	end
	
	game.Players:FindFirstChild("MSpinMan").leaderstats.Clicks.Value += 1
end)

tool.Deactivated:Connect(function()
	print("deactivated")
	holding = false
	tool:ScaleTo(1)
end)
